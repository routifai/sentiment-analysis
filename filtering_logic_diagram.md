### **3. Scoring Algorithm**

def calculate_feedback_score(text):
    keywords = extract_keywords(text)
    pattern_matches = count_pattern_matches(text)
    
    # Weighted scoring
    keyword_weight = 1.0
    pattern_weight = 0.5

    return keywords, pattern_matches, score
